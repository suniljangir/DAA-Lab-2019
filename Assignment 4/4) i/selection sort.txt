int a[7];
int select_min(int l,int n)
{
int min,minloc,i;
min=a[l];
minloc=l;
for(i=l+1;i<=n;i++)
{ if(min>a[i])
  {min=a[i];
   minloc=i;
  }
}
return(minloc);
}
